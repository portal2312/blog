Database

Dump:

pg_dump dbname > dumpfile

Restore:

pg_dump dbname < dumpfile

Table

Dump:

pg_dump --username=username --dbname=dbname --table=table_name > dbname.table.sql

Restore:

psql --username=username --dbname=dbname --table=table_name --file=dbname.table.sql

Column

Add:

ALTER TABLE MY_TABLE add column "COLUMN" varchar(50) NULL;

Remove:

ALTER TABLE MY_TABLE drop column "COLUMN";

Rename:

ALTER TABLE "TABLE" rename column "TABLE_COLUMN_old" to "TABLE_COLUMN_new";

Drop unique:

drop icontras "COLUMN"

Create "COLUMN" index on "TABLE":

CREATE INDEX "TABLE_COLUMN" ON "TABLE" ("COLUMN");

Add foreign key, "MAIL" user_id -> "USER"의 id:

ALTER TABLE "MAIL" add contraint "MAIL_user_id_fkey" add foreign key (user_id) references "USER"(id) DEFERRABLE INITIALLY DEFERRED;

Remove foreign key:

ALTER TABLE "TABLE" drop constraint "TABLE_column_id_fkey";
