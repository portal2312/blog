Without Foreign Key

import time
from django.db.models.sql.datastructures import Join
from django.db.models.fields.related import ForeignObject
from django.db.models.options import Options
from django.db.models.expressions import Col
from django.db.models import (
    BigIntegerField,
    Max,
    Sum,
    Max,
    Value as V,
    #, F
)
from django.db.models.functions import Concat
from control.models import *


def test():
    """
    SELECT
        "SOURCE_COLLECTOR_MAP"."id",
        "SOURCE_COLLECTOR_MAP"."source_id",
        "SOURCE_COLLECTOR_MAP"."collector_id",
        "SOURCE_COLLECTOR_MAP"."state",
        "SOURCE_COLLECTOR_MAP"."is_active",
        "SOURCE_COLLECTOR_MAP"."ctime",
        "SOURCE_COLLECTOR_MAP"."mtime",
        "SOURCE_STAT_SIZE"."raw_size" AS "aaa"
    FROM "SOURCE_COLLECTOR_MAP"
    LEFT OUTER JOIN "SOURCE_STAT_SIZE"
    ON (
        "SOURCE_COLLECTOR_MAP"."source_id" = "SOURCE_STAT_SIZE"."source_id" AND
        "SOURCE_COLLECTOR_MAP"."collector_id" = "SOURCE_STAT_SIZE"."collector_id"
    )
    LIMIT 21;
    :return:
    """

    jf = ForeignObject(
        to=SourceStatSize,
        on_delete=lambda: x,
        from_fields=['collector_id', 'source_id'],
        to_fields=['collector_id', 'source_id'],
        rel=None,
        related_name=None
    )

    jf.opts = Options(SourceCollectorMap._meta)
    jf.opts.model = SourceCollectorMap
    jf.get_joining_columns = lambda: (
        ("source_id", "source_id"),
        ("collector_id", "collector_id"),
    )

    j = Join(
        SourceStatSize._meta.db_table,
        SourceCollectorMap._meta.db_table,
        'SOURCE_STAT_SIZE',
        'INNER JOIN',
        jf,
        True
    )

    q = SourceCollectorMap.objects.filter(
    ).annotate(
        aaa=Col(
            SourceStatSize._meta.db_table,
            SourceStatSize._meta.get_field('raw_size'),
            output_field=BigIntegerField()
        )
    ).values(
    )
    q.query.join(j)
    return q
