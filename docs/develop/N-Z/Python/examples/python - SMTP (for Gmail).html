import smtplib
from email.mime.text import MIMEText

domain = 'smtp.gmail.com'
port = 587

smpt = smtplib.SMTP(domain, port)

# Set TLS.
smpt.starttls()
# (220, b'2.0.0 Ready to start TLS')

username = 'Gmail account'
passwd = 'Gmail account App passwd'
smpt.login(username, passwd)
# (235, b'2.7.0 Accepted')

# Create message.
msg = MIMEText('description message')
msg['Subject'] = 'subject'
print(msg.as_string())
# Content-Type: text/plain; charset="us-ascii"
# MIME-Version: 1.0
# Content-Transfer-Encoding: 7bit
# Subject: subject
# 
# description message

# Send test mail from self to self.
from_ = username
to = username
smpt.sendmail(from_, to, msg.as_string())
# {}

# Close session.
smpt.quit()
# (221, b'2.0.0 closing connection g2sm3875758pfb.95 - gsmtp')
