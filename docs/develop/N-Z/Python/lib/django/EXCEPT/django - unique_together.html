unique_together 사용시 기존 field 의 unique=True 는 반드시 제거하기

제거하지 않는 경우 아래 DateTimeField(null=True)시 동작 안함:

class Server(models.Model):
    """
    검사 대상 서버

    target_port
    : 검사 대상 포트 범위 ex) 1,2

    is_delete
    : delete() method 에 의한 삭제가 아닌 is_delete 로 표현하기

    delete_time
    : is_delete 를 이용한 삭제 처리시 unique 관련 이슈로 인해 사용하기
    : 반드시 is_delete=True 시 delete_time=int(time.time()) 하기

    - models.PositiveIntegerField
        - DatetimeField 사용하지 않는 이유
            - null=True 시 unique 가 되지 않는다
            - default 의 옳바른것 값이 없다
    """
    TYPES_DEFAULT = 'general'
    TYPES_CHOICES = (
        (TYPES_DEFAULT, '일반'),
        ('security', '보안'),
        ('network', '네트워크'),
        ('etc', '기타'),
    )
    name = models.CharField(max_length=200)
    ip = models.CharField(max_length=15)
    types = models.CharField(max_length=32, choices=TYPES_CHOICES, default=TYPES_DEFAULT)
    scanner = models.ForeignKey(PortScanner, on_delete=models.SET_NULL, null=True)
    target_port = models.CharField(max_length=200, default='1-65535')
    create_date = models.DateTimeField(default=timezone.now)
    update_date = models.DateTimeField(auto_now=True)
    is_delete = models.BooleanField(default=False)
    delete_time = models.PositiveIntegerField(default=0)
    # 관리자 관련 정보는 나중에 추가 필요


    def __str__(self):
        return f'{self.name}({self.ip})'

    class Meta:
        ordering = ('name', '-create_date')
        indexes = [
            models.Index(fields=('name', 'is_delete', 'delete_date')),
            models.Index(fields=('ip', 'is_delete', 'delete_date')),
        ]
        unique_together = (
            ('name', 'is_delete', 'delete_date'),
            ('ip', 'is_delete', 'delete_date'),
        )
