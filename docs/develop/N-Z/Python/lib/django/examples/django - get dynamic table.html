# 동적 테이블 찾기
# 동적 filters 만들기
#
import json
import time
from control.models import UCEMSConnectResult
from control.deco import json_encode
from django.utils.decorators import method_decorator
from django.apps import apps
from ..baseview import View


class InfomationView(View):
    app_label = 'control'
    date_format = '%Y-%m-%d %H:%M'

    def __init__(self):
        self.now = int(time.time())
        self.now_str = time.strftime(self.date_format)

    def get_attname_internal_type(self, model):
        fields = model._meta.get_fields()
        result = dict(map(lambda x: (x.get_attname(), x.get_internal_type()), fields))
        return result

    def get_filters(self, request_form_data, attname_internal_type, priorities={}, date_format='%Y-%m-%d %H:%M'):
        """
        Get Model Default Filters

        MUST:
        - date: request_form_data(=request.POST)에서 %s_start, %s_limit 명시하기
          - example: request.POST.get('ctime_start'), request.POST.get('ctime_limit')

        :param request_form_data: request.POST
        :param attname_internal_type: {'id': 'AutoField', 'name': 'CharField', ...}
        :param priorities: {'keys': {}, 'values': {}}
        :param date_format: '%Y-%m-%d %H:%M'
        :return: {'id': 1, 'name__icontains': 'admin', 'ctime__gte': 10000, 'ctime__lte': 20000, ...}
        """
        result = {}

        if not priorities:
            # default
            priorities = {
                'keys': {},
                'values': {}
            }

        for attname in attname_internal_type:
            # value = attname 로 먼저 priorities 시도 이후 없다면 request_form_data 에서 값 가져오기
            value = priorities.get('values').get(attname, request_form_data.get(attname))

            # internal_type = '*Field' 유형의 값, field 의 scheme 정보
            internal_type = attname_internal_type[attname]

            # __key = 기본값 설정하기 (다양한 유형으로 변경 될 수 있다)
            _key = priorities.get('keys').get(attname, '%s') % attname

            if attname in ('ctime', 'mtime', 'stime', 'etime', 'init_time'):
                if internal_type == 'IntegerField':
                    # MUST '%s_start' 명시하기
                    value = request_form_data.get('%s_start' % attname)
                    if value:
                        value = int(time.mktime(time.strptime(value, date_format)))
                        _key = _key == attname and '%s__gte' % attname or _key  # default
                        result[_key] = value

                    # MUST '%s_limit' 명시하기
                    value = request_form_data.get('%s_limit' % attname)
                    if value:
                        value = int(time.mktime(time.strptime(value, date_format)))
                        _key = _key == attname and '%s__lte' % attname or _key  # default
                        result[_key] = value
                    continue

            if value in (None, ''):
                continue

            if internal_type == 'BooleanField':
                if type(value) is not int:
                    # value in ('0', '1', 'false', 'true')
                    value = json.loads(value)
                value = bool(value)
            elif internal_type in ('CharField', 'TextField'):
                _key = _key == attname and '%s__icontains' % attname or _key  # default
            elif internal_type == 'IntegerField' and type(value) is not int:
                value = int(value)

            result[_key] = value

        return result

    @method_decorator(json_encode)
    def read(self, request):
        category = request.POST.get('category')
        said = request.POST.getlist('said')
        # sort = request.POST.get('sort')

        if not category:
            return

        model = apps.get_model(self.app_label, 'ucems%sinfo' % (category,))
        attname_internal_type = self.get_attname_internal_type(model)
        request_form_data = request.POST
        priorities = {
            'keys': {
                'said': '%s__in'
            },
            'values': {
                'said': said
            }
        }
        filters = self.get_filters(request_form_data, attname_internal_type, priorities)
        sorts = ['ctime']
        query_set = model.objects.filter(**filters).values().order_by(*sorts)
        result = list(query_set)
        return result
