install

  • yum install git-core

config

autocomplite

  • 안될 경우 bash 명령어 실행 이후 다시 시도.

profile



git config --global user.name "John Doe"
git config --global user.email johndoe@example.com
git config --core.editor vim
git config --list

For Windows

install

  • install .exe 실행

git config --global

# git config --global -e 해서 등록해서 사용하기
glog = log --graph --pretty=format:'%ci %Cgreen(%cr)%Creset %C(auto)%h%Creset %cn %C(auto)%d%Creset -- %s' --abbrev-commit --decorate --all

# 간편 명령어
git log --oneline --graph --all --decorate
