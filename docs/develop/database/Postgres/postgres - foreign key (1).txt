# **Foreign key**

- 전체 예제

alter table "ACCOUNT_USER_MAP" add constraint account_user_map_key unique (account_id, user_id);

ALTER TABLE products ADD FOREIGN KEY (product_group_id) REFERENCES product_groups;

alter table "ACCOUNT_HISTORY" add foreign key ("ACCOUNT_HISTORY_admin_id") references "USER_id";

CREATE INDEX "HOSTLOG_ctime" ON "HOSTLOG" ("ctime");



## Add

- "MAIL"의 user_id 가 "USER"의 id를 참조하게 하기

alter table "MAIL"  add contraint "MAIL_user_id_fkey" add foreign key (user_id) references "USER"(id) DEFERRABLE INITIALLY DEFERRED;


## Drop

- 삭제시 contraint 명으로 삭제해야 한다

alter table "TEST" drop constraint "TEST_admin_id_fkey";


## Rename constraint

alter table “ACCOUNT_PASSWD_HISTORY” rename column “ACCOUNT_PASSWD_HISTORY_tmp” to "ACCOUNT_PASSWD_HISTORY_script_id"
