# **boot full**

df -h  # check current boot size

uname -a  # check using (new version) kernel

...3.10.0-327.36.3.el7.x86_64...

rpm -qa | grep kernel | sort  # show installed kernel list

yum remove kernel-x.x.x-x.x.x86_64  # remove (old version) kernel

df -h  # check current boot size

yum upgrade

vi /etc/yum.conf

installonly_limit=2

:wq
