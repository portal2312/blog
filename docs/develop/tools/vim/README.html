Tips

﻿multi file open 시 file view movement command.

Options Desc
:p prev
:n next

find 이용한 multi fild open.

$ find . -name "*.txt" -exec vim {} +

one line > multi line.

qq (기록 모드로 진입)
이동 및 기록 (수정내용 메크로 만들기)
q (기록 모드 종료)
5 (=반복숫자 입력)
@
q

multi line > one line.

visual-block 잡은 후 shift + j

vi 를 vim 으로 교체

su -
cd /usr/bin
mv -v vi vi_bak
ln -s vim vi

환경설정

vi /etc/vimrc

...
set bg=dark    " background=dark, font color 전반적으로 변경됨.
set nu    " number, file open 시 row 번호 표현
...

ArchLinux

처음에 vi, vim 이 설치 안되있으므로 pacman 으로 설치후 진행.

환경설정은 /etc/vimrc 에 그대로 기입하면 됨.

vimrc


set autoindent " 자동 들여쓰기
set autowrite " 다른 파일로 넘어갈 때 자동 저장
set autoread  " 작업 중인 파일 외부에서 변경됬을 경우 자동으로 불러옴
set bg=dark    " background color
set bs=eol,start,indent
set expandtab  " Tab-key space
set history=256
set hlsearch   " 검색어 하이라이팅
set incsearch
set laststatus=2 " 상태바 표시 항상
set nu         " number(line-number)
set ruler  " 현재 커서 위치 표시
set sts=4     " st select
set sw=1      " 스크롤바 너비
set shiftwidth=4 " 자동 들여쓰기 너비 설정
set showmatch    " 일치하는 괄호 하이라이팅
set smartcase    " 검색시 대소문자 구별
set smarttab
set smartindent
set scrolloff=2
set softtabstop=4
set statusline=\ %<%l:%v\ [%P]%=%a\ %h%m%r\ %F\
set ts=4      " tag select
set tabstop=4
set wildmode=longest,list

" 구문 강조 사용
if has("syntax")
 syntax on
endif

" set cindent   " C언어 자동 들여쓰기

" 마지막으로 수정된 곳에 커서를 위치함
" au BufReadPost *
" \ if line("'\"") > 0 && line("'\"") <= line("$") |
" \ exe "norm g`\"" |
" \ endif
" " 파일 인코딩을 한국어로
" if $LANG[0]=='k' && $LANG[1]=='o'
" set fileencoding=korea
" endif
